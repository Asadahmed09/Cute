# 2. The Minimax Function
def minimax(node, is_maximizing_player):
    # BASE CASE: If the node is just a number (a leaf), return the number
    if isinstance(node, int):
        return node

    # Get the children of the current node from our dictionary
    children = game_tree[node]

    # MAXIMIZER'S TURN
    if is_maximizing_player:
        best_score = -999  # Start with an impossibly low score

        for child in children:
            # Time travel down the tree, passing the turn to the Minimizer (False)
            score = minimax(child, False)
            best_score = max(best_score, score)

        return best_score

    # MINIMIZER'S TURN
    else:
        best_score = 999  # Start with an impossibly high score

        for child in children:
            # Time travel down the tree, passing the turn to the Maximizer (True)
            score = minimax(child, True)
            best_score = min(best_score, score)

        return best_score


optimal_value = minimax('A', True)

#MIN MAX GRID 
# The 1D Board limits
MIN_BOARD = 0
MAX_BOARD = 4
GOAL = 4


# 1. The Heuristic Function
def evaluate_board(max_pos, min_pos):
    if max_pos == GOAL:
        return 100  # Max won!
    if max_pos == min_pos:
        return -100  # Max was caught by Min!

    # HEURISTIC: If the game isn't over, reward Max for being closer to the goal
    return max_pos * 10


# 2. Minimax with Depth Limit
def minimax(max_pos, min_pos, depth, is_max_turn):
    # BASE CASE 1: Game Over (Win/Lose)
    if max_pos == GOAL or max_pos == min_pos:
        score = evaluate_board(max_pos, min_pos)
        return score, None

    # BASE CASE 2: Depth Limit Reached (Use Heuristic!)
    if depth == 0:
        score = evaluate_board(max_pos, min_pos)
        print(f"      [Leaf Reached] Max at {max_pos}, Min at {min_pos} -> Heuristic Score: {score}")
        return score, None

    moves = [1, 0, -1]  # Right, Stay, Left

    if is_max_turn:
        best_score = -999
        best_move = 0
        for move in moves:
            new_max = max_pos + move
            if new_max < MIN_BOARD or new_max > MAX_BOARD: continue  # Don't fall off board

            score, _ = minimax(new_max, min_pos, depth - 1, False)
            if score > best_score:
                best_score = score
                best_move = move
        return best_score, best_move

    else:  # Min's Turn
        best_score = 999
        best_move = 0
        for move in moves:
            new_min = min_pos + move
            if new_min < MIN_BOARD or new_min > MAX_BOARD: continue

            score, _ = minimax(max_pos, new_min, depth - 1, True)
            if score < best_score:
                best_score = score
                best_move = move
        return best_score, best_move


# 3. Simulate 3 Game States (Max -> Min -> Max)
current_max = 0
current_min = 3
SEARCH_DEPTH = 3

print("--- STARTING SIMULATION ---")
print(f"Initial State: Max at {current_max}, Min at {current_min}, Goal at {GOAL}\n")

for turn in range(3):
    is_max = (turn % 2 == 0)
    player_name = "Max" if is_max else "Min"

    print(f"--- Turn {turn + 1}: {player_name}'s Turn ---")

    # Ask the AI for the best move
    best_score, chosen_move = minimax(current_max, current_min, SEARCH_DEPTH, is_max)

    # Apply the move to the real board
    if is_max:
        current_max += chosen_move
    else:
        current_min += chosen_move

    print(f"> {player_name} chooses to move: {chosen_move}")
    print(f"> Current Board: Max is at {current_max}, Min is at {current_min}\n")

    # Stop simulation if game naturally ends
    if current_max == GOAL or current_max == current_min:
        print("Game Over!")
        break

#tic tac toe | GRID | min max + alpha beta
import math

# Constants
X_PLAYER = 'X'  # AI (Max)
O_PLAYER = 'O'  # Human (Min)
EMPTY = ' '


class TicTacToeAI:
    def __init__(self):
        self.board = [EMPTY] * 9  # 3x3 board represented as a 1D list (indices 0-8)
        self.pruned_count = 0
        self.max_depth_explored = 0

    def print_board(self):
        for i in range(3):
            row = f" {self.board[i * 3]} | {self.board[i * 3 + 1]} | {self.board[i * 3 + 2]} "
            print(row)
            if i < 2:
                print("---|---|---")
        print()

    def get_empty_cells(self):
        return [i for i, cell in enumerate(self.board) if cell == EMPTY]

    def is_winner(self, player):
        # All possible winning lines
        win_states = [
            [0, 1, 2], [3, 4, 5], [6, 7, 8],  # Rows
            [0, 3, 6], [1, 4, 7], [2, 5, 8],  # Columns
            [0, 4, 8], [2, 4, 6]  # Diagonals
        ]
        for state in win_states:
            if self.board[state[0]] == self.board[state[1]] == self.board[state[2]] == player:
                return True
        return False

    def is_draw(self):
        return EMPTY not in self.board

    def evaluate(self, depth):
        # AI wants to win as FAST as possible, and lose as SLOWLY as possible.
        # We factor in the 'depth' to penalize longer paths.
        if self.is_winner(X_PLAYER):
            return 10 - depth
        elif self.is_winner(O_PLAYER):
            return -10 + depth
        else:
            return 0

    def alpha_beta(self, depth, alpha, beta, is_maximizing):
        # Track the deepest part of the tree we reach during this search
        self.max_depth_explored = max(self.max_depth_explored, depth)

        score = self.evaluate(depth)

        # BASE CASE: If someone won, or board is full, return the score
        if score != 0 or self.is_draw():
            return score

        # AI's Turn (Max)
        if is_maximizing:
            best_score = -math.inf
            for cell in self.get_empty_cells():
                self.board[cell] = X_PLAYER  # Make move
                score = self.alpha_beta(depth + 1, alpha, beta, False)
                self.board[cell] = EMPTY  # Undo move

                best_score = max(best_score, score)
                alpha = max(alpha, best_score)

                # THE PRUNING SCISSORS
                if alpha >= beta:
                    self.pruned_count += 1
                    break
            return best_score

        # Human's Turn (Min)
        else:
            best_score = math.inf
            for cell in self.get_empty_cells():
                self.board[cell] = O_PLAYER  # Make move
                score = self.alpha_beta(depth + 1, alpha, beta, True)
                self.board[cell] = EMPTY  # Undo move

                best_score = min(best_score, score)
                beta = min(beta, best_score)

                # THE PRUNING SCISSORS
                if alpha >= beta:
                    self.pruned_count += 1
                    break
            return best_score

    def find_best_move(self):
        # Reset tracking stats for the new turn
        self.pruned_count = 0
        self.max_depth_explored = 0

        best_score = -math.inf
        best_move = -1
        alpha = -math.inf
        beta = math.inf

        for cell in self.get_empty_cells():
            self.board[cell] = X_PLAYER
            score = self.alpha_beta(0, alpha, beta, False)
            self.board[cell] = EMPTY

            if score > best_score:
                best_score = score
                best_move = cell

            # Root level alpha update
            alpha = max(alpha, best_score)

        return best_move

    def play(self):
        print("--- Alpha-Beta Tic-Tac-Toe! ---")
        print("You are 'O'. Positions are 0-8, reading left-to-right, top-to-bottom.\n")
        self.print_board()

        while not self.is_winner(X_PLAYER) and not self.is_winner(O_PLAYER) and not self.is_draw():

            # --- AI TURN (Max) ---
            print("AI is thinking...")
            best_move = self.find_best_move()
            self.board[best_move] = X_PLAYER

            print(f"> AI (X) played position {best_move}")
            print(f"  [Stat] Max Depth Explored: {self.max_depth_explored}")
            print(f"  [Stat] Branches Pruned: {self.pruned_count}\n")
            self.print_board()

            if self.is_winner(X_PLAYER) or self.is_draw():
                break

            # --- HUMAN TURN (Min) ---
            valid_move = False
            while not valid_move:
                try:
                    move = int(input("Your turn (O). Enter position (0-8): "))
                    if move >= 0 and move <= 8 and self.board[move] == EMPTY:
                        self.board[move] = O_PLAYER
                        valid_move = True
                    else:
                        print("Invalid or taken position! Try again.")
                except ValueError:
                    print("Please enter a valid number (0-8).")
            print()
            self.print_board()

        # Game Over Logic
        if self.is_winner(X_PLAYER):
            print("AI Wins! (Mathematically guaranteed 😉)")
        elif self.is_winner(O_PLAYER):
            print("Human Wins! (Wait, my code is broken if you see this!)")
        else:
            print("It's a Draw! Perfectly played on both sides.")

    game = TicTacToeAI()
    game.play()
